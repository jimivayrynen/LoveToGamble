3    object CardValueHelper:
4    
5      // Muuntaa kortin numeerseen arvon
6      def cardValue(card: PlayingCard): Int = card.value match
7        case "Ace" => 1
8        case "Jack" => 11
9        case "Queen" => 12
10       case "King" => 13
11       case n => n.toIntOption.getOrElse(0)
12   
13   object CardValidator:
14   
15     // kaikki mahdolliset kombinaatiot
16     def findValidTakes(playerCard: PlayingCard, tableCards: List[PlayingCard]): List[List[PlayingCard]] =
17       val target = CardValueHelper.cardValue(playerCard)
18   
19       val allCombos = tableCards.toSet.subsets().filter(_.nonEmpty).map(_.toList).toList
20   
21       val validCombos = allCombos.filter(c => c.map(CardValueHelper.cardValue).sum == target)
22   
23       def nonOverLapping(remaining: List[List[PlayingCard]], acc: List[PlayingCard]): List[List[PlayingCard]] =
24         remaining match
25           case Nil => List(acc)
26           case head :: tail =>
27             if head.forall(!acc.contains(_)) then
28               val withThis = nonOverLapping(tail, acc ++ head)
29               val withoutThis = nonOverLapping(tail, acc)
30               withThis ++ withoutThis
31             else
32               nonOverLapping(tail, acc)
33   
34       val merged = nonOverLapping(validCombos,List()).filter(_.nonEmpty)
35       val sameValue = tableCards.filter(_.value == playerCard.value)
36       val valueMatch = if sameValue.nonEmpty then List(sameValue) else Nil
37   
38       merged ++ valueMatch
39   
40     // Onko siirto validi eli voiko joko ottaa tai pelata pöytään
41     def validateMove(playerCard: PlayingCard, tableCards: List[PlayingCard]): Boolean =
42       findValidTakes(playerCard, tableCards).nonEmpty || canPlayCard(playerCard, tableCards)
43   
44     // Kortin pelaaminen pöytään on aina sallittua (jos ei oteta mitään)
45     def canPlayCard(playerCard: PlayingCard, tableCards: List[PlayingCard]): Boolean = true

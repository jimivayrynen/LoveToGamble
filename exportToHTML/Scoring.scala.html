3    class Scoring:
4    
5      // päivitetään pelaajan pisteet
6      def updateScore(player: Player): Int =
7        val aces = player.collectedCards.count(_.value == "Ace") // ässät
8        val tenOfHearts = player.collectedCards.count(c => c.value == "10" && c.suit == "Diamonds") * 2 //hertta kymppi
9        val mokkiPoints = player.mokkiCount // mökit
10       aces + tenOfHearts + mokkiPoints
11   
12   
13     // määrittää voittajan
14     def finalizeScore(players: List[Player]): Unit =
15       players.foreach(player =>
16         val roundPoints = updateScore(player)
17         player.points += roundPoints
18         player.mokkiCount = 0)
19   
20       //Eniten kortteja
21       val mostCards = players.maxByOption(_.collectedCards.size)
22       // Eniten patoja
23       val mostSpades = players.maxByOption(_.collectedCards.count(_.suit == "Spades"))
24   
25       mostCards.foreach(_.points += 1)
26       mostSpades.foreach(_.points += 1)
27   
28   
29   end Scoring

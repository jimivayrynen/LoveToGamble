9    class Game(var players: List[Player]):
10   
11     var currentPlayerIndex = 0
12     val scoring = new Scoring()
13     val board = new Board()
14     var tableCards: List[PlayingCard] = List() // Kortit pöydällä
15     var lastTaker: Option[Player] = None
16   
17     // aloittaa pelin
18     def startGame(): Unit =
19       board.shuffleDeck()
20       players.foreach(p =>
21         val cards = board.dealCards(4)
22         p.sethand(cards))
23       board.initTable()
24   
25     def playCard(card: PlayingCard, selected: List[PlayingCard]): Boolean =
26       if currentPlayer.hand.contains(card) then
27         val takes = CardValidator.findValidTakes(card, board.tableCards)
28   
29         currentPlayer.playCard(card)
30   
31         if takes.nonEmpty then
32           val taken = takes.head
33           board.tableCards = board.tableCards.filterNot(taken.contains)
34           currentPlayer.collectedCards = (card +: taken) ++ currentPlayer.collectedCards
35           lastTaker = Some(currentPlayer)
36   
37           if board.tableCards.isEmpty then
38             currentPlayer.mokkiCount += 1
39         else
40           board.addCardToTable(card)
41   
42   
43         // Jos kaikkien kortit pelattu mutta peli ei ole vielä ohi
44         if players.forall(_.hand.isEmpty) then
45           lastTaker.foreach(_.collectedCards ++= board.tableCards)
46           board.tableCards = Nil
47           scoring.finalizeScore(players)
48           if board.deck.cardsLeft == 0 && (players.forall(_.hand.isEmpty)) then
49             startNextRound()
50   
51         true
52       else false
53   
54   
55     def currentPlayer: Player = players(currentPlayerIndex)
56   
57     // määrittää pelin lopun
58     def isGameOver: Boolean =
59       players.exists(_.points >= 16)
60   
61     def winner: Option[Player] =
62       if players.exists(_.points >= 16) then
63         Some(players.maxBy(_.points))
64       else None
65   
66   
67     def endGame():Unit =
68       scoring.finalizeScore(players)
69   
70     def startNextRound(): Unit =
71       players.foreach(_.hand = List())
72   
73       val newDeckCards = players.flatMap(_.collectedCards) ++ board.tableCards
74       players.foreach(_.collectedCards = List())
75       board.resetDeck(newDeckCards)
76       board.tableCards = List()
77   
78       players.foreach(p => p.sethand(board.dealCards(4)))
79       board.initTable()
80       currentPlayerIndex = 0
81   
82   
83     // päivittää pelin
84     private def updateGameState(player: Player): Unit =
85       scoring.updateScore(player)
86       board.showTable()
87   
88     // seuraavan pelaajan vuoro
89     def switchTurn(): Unit =
90       if players.forall(_.hand.isEmpty) then
91         if board.deck.cardsLeft == 0 then
92           endGame()
93         else
94           startNextRound()
95       else
96         currentPlayerIndex = (currentPlayerIndex + 1) % players.size
97   
98       // Täydennetään seuraavan pelaajan käsi
99       val player = currentPlayer
100      val missing = 4 - player.hand.size
101      if missing > 0 then
102        val newCards = board.dealCards(missing)
103        player.hand = player.hand ++ newCards
104  
105      player match
106        case ai: ComputerPlayer =>
107          Platform.runLater(
108            Main.showTemoraryMessage("Tietokone pelaa...", Color.LightGreen))
109            Future {
110              Thread.sleep(2000)
111              val (card, take) = ai.decideMove(board.tableCards)
112              Platform.runLater {
113                playCard(card, take)
114                switchTurn()
115                Main.updateUI()
116              }
117            }
118        case _ =>
119          Platform.runLater(Main.updateUI())

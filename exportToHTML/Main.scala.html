41     def showStartMenu(): Unit =
42       mainRoot.children.clear()
43   
44       val title = new Text("Mökki Kasino"):
45         font = Font(32)
46         fill = Gold
47         layoutX = 330
48         layoutY = 50
49   
50       val includeAI = new ComboBox[String](Seq("Ei tekoälyä", "Lisää tietokonepelaaja")):
51         value = "Ei tekoälyä"
52         layoutX = 500
53         layoutY = 100
54   
55       val playerCountBox = new ComboBox[Int](Seq(1, 2, 3, 4)):
56         value = 1
57         layoutX = 400
58         layoutY = 100
59   
60       val nameFields = new VBox:
61         layoutX = 360
62         layoutY = 150
63         spacing = 10
64   
65       def updateNameFields(count: Int): Unit =
66         nameFields.children = (1 to count).map( i => new TextField {promptText = s"Pelaaja $i"})
67   
68       updateNameFields(playerCountBox.value())
69       playerCountBox.onAction = _ => updateNameFields(playerCountBox.value())
70   
71       val startButton = new Button("Aloita peli"):
72         layoutX = 420
73         layoutY = 320
74         onAction = _ =>
75           val names = nameFields.children.toList.collect { case node: javafx.scene.control.TextField => node.getText}.filter(_.nonEmpty)
76           if names.size == playerCountBox.value() then
77             val players = names.map(n => new Player(n)).toList
78             val finalPlayers = if includeAI.value() == "Lisää tietokonepelaaja" then
79               players :+ new ComputerPlayer("Tietokone")
80             else
81               players
82             game = new Game(finalPlayers)
83             game.startGame()
84             updateUI()
85           else
86             showTemoraryMessage("Syötä kaikkien nimet", Color.Red)
87   
88       mainRoot.children ++= Seq(title, playerCountBox, nameFields, startButton, includeAI)

113          width = 600
114          height = 200
115          fill = DarkOliveGreen
116          arcWidth = 20
117          arcHeight = 20

126              font = Font(20)

129              font = Font(18)

131            new Label(s"Kortteja jäljellä: ${game.board.deck.cardsLeft}"):
132              font = Font(16)
133              textFill = White

138        val playerScores = new HBox:
139          layoutX = 100
140          layoutY = 420

153          val winnerName = game.winner.map(_.name).getOrElse("Tuntematon")
154          val winText = new Text(300, 250, s"$winnerName voitti pelin!"):
155            font = Font(30)
156            fill = Gold
157            effect = new DropShadow(radius = 10, color = Black)
158  
159          val newGameButton = new Button("Uusi peli?"):
160            layoutX = 400
161            layoutY = 300
162            onAction = _ => showStartMenu()
163  
164          mainRoot.children ++= Seq(winText, newGameButton)
165        else renderCards()
166

170    // piirtää pelaajien ja pöydällä olevien kortit
171    def renderCards(): Unit =
172      //käden kortin valinta
173      game.currentPlayer.hand.zipWithIndex.foreach( (card, i) =>
174        val cardView = createCardImageView(200 + i * 80, 500, card)
175        cardView.onMouseEntered = _ => cardView.translateY = -10
176        cardView.onMouseExited = _ => cardView.translateY = 0
177  
178        if selectedHandCard.contains(card) then
179          cardView.effect = new DropShadow(10, LightBlue)
180  
181        cardView.onMouseClicked = _ =>
182          selectedHandCard = Some(card)
183          updateUI()
184        mainRoot.children += cardView)
185  
186      //pöytäkorttien valinta
187      game.board.tableCards.zipWithIndex.foreach( (card, i) =>
188        val cardView = createCardImageView(150 + i *40, 100, card)
189  
190        if selectedTableCards.contains(card) then
191          cardView.effect = new DropShadow(10, Yellow)
192  
193        cardView.onMouseClicked = _ =>
194          if selectedTableCards.contains(card) then
195            selectedTableCards -= card
196          else
197            selectedTableCards += card
198          updateUI()
199        mainRoot.children += cardView)
200  
201      //pelaa nappi, toimii jos käden kortti on valittu
202      val playButton = new Button("Pelaa"):
203        layoutX = 50
204        layoutY = 550
205        disable = selectedHandCard.isEmpty
206        onAction = _ =>
207          selectedHandCard.foreach( handcard =>
208            val succes = game.playCard(handcard, selectedTableCards.toList)
209            if succes then
210              selectedHandCard = None
211              selectedTableCards = Set()
212              game.switchTurn()
213              updateUI()
214            else
215              showTemoraryMessage("Virheellinen siirto!", Color.Red))
216  
217      mainRoot.children += playButton
218

227    //Luo kortin kuvaesityksen GUI:hin annetuilla sijainneilla
228    def createCardImageView(xPos: Double, yPos: Double, card: PlayingCard): ImageView =
229      val path = s"/cards/${card.imagePath}"
230      val stream = getClass.getResourceAsStream(s"/cards/${card.imagePath}")
231  
232      if stream == null then
233      throw new RuntimeException(s"Kuvaa ei löytynyt: $path")
234  
235      val image = new Image(stream)
236      new ImageView(image):
237        fitWidth = 50
238        fitWidth = 70
239        preserveRatio = true
240        this.x = xPos
241        this.y = yPos
242  
243    //Näyttää hetkellisen virheilmoituksen
244    def showTemoraryMessage(text: String, color: Color): Unit =
245      val msg = new Label(text):
246        layoutX = 350
247        layoutY = 350
248        font = Font(20)
249        textFill = color
250        effect = new DropShadow(radius = 10, color = Black)
251  
252      mainRoot.children += msg
253  
254      val fadeOut = new javafx.animation.FadeTransition(javafx.util.Duration.seconds(2), msg)
255      fadeOut.setFromValue(1.0)
256      fadeOut.setToValue(0.0)
257      fadeOut.setOnFinished(_ => mainRoot.children -= msg)
258      fadeOut.play()

13   object Main extends JFXApp3:
14   
15     var currentPlayerIndex = 0
16     val game = new Game(List(new Player("Player 1"), new Player("Player 2")))
17   
18     def start() =
19   
20       stage = new JFXApp3.PrimaryStage:
21         title = "Mökki Kasino"
22         width = 600
23         height = 450
24   
25       val root = Pane()
26   
27       val scene = Scene(parent = root)
28       stage.scene = scene
29   
30       // komponenteille oma lisäys
31       addComponentsToRoot(root)
32   
33       // luodaan pelaajien kortit ja päivitetään pelitilanne
34       updateGameState(root)
35   
36     // kortin kikkaus
37     def handleCardClick(e: MouseEvent, card: PlayingCard, root: Pane): Unit =
38       println(s"Klikattu korttia: $card")
39       game.players(game.currentPlayerIndex).playCard(card)
40   
41       if (CardValidator.validateMove(card, game.board.tableCards)) then
42         game.board.addCardToTable(card)
43         game.scoring.updateScore(game.players(game.currentPlayerIndex))
44         game.board.showTable()
45   
46         game.switchTurn()
47         updateGameState(root)
48       else
49         println("Invalid move, try again!")
50   
51     // päivitetään peliin liittyvät tiedot
52     def updateGameState(root: Pane): Unit =
53       root.children.clear()
54   
55       // komponentit
56       addComponentsToRoot(root)
57   
58       val currentPlayer = game.players(game.currentPlayerIndex)
59       currentPlayer.hand.zipWithIndex.foreach( (card, i) =>
60         val cardView = createCardImageView(120 + i * 100, 400, card)
61         root.children += cardView
62         cardView.onMouseClicked = (e: MouseEvent) => handleCardClick(e, card, root))
63   
64   
65     // lisätään kaikki kerrallaan
66     def addComponentsToRoot(root: Pane): Unit =
67       //pelipöytä
68       val tableRectangle = new Rectangle {
69         x = 100
70         y = 50
71         width = 400
72         height = 200
73         fill =LightGray
74       }
75   
76       // pelitilan tekstit
77       val gameInfo = new Text {
78         x = 10
79         y = 20
80         text = s"Pelaaja ${currentPlayerIndex + 1} vuoro"
81         style = "-fx-font-size: 14px"
82       }
83   
84       root.children ++= List(tableRectangle, gameInfo)
85   
86     def createCardImageView(xPos: Double, yPos: Double, card: PlayingCard): ImageView =
87       val cardImage = new Image(getClass.getResourceAsStream(s"/resources/cards/${card.imagePath}"))
88       val cardImageView = new ImageView(cardImage)
89       cardImageView.x = xPos
90       cardImageView.y = yPos
91       cardImageView
92   
93   
94   
95   end Main

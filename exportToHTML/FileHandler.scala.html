23     def loadGame(path: String): Game =
24       val lines = Files.readAllLines(Paths.get(path), StandardCharsets.UTF_8).asScala.toList
25   
26       var players: List[Player] = List()
27       var tableCards: List[PlayingCard] = List()
28       var deckCards: List[PlayingCard] = List()
29       var turnPlayerName = ""
30   
31       lines.foreach{
32         case line if line.startsWith("PLAYER:") =>
33           val parts = line.stripPrefix("PLAYER:").split(":")
34           val name = parts(0)
35           val points = parts(1).toInt
36           val hand = if parts.length > 2 && parts(2).nonEmpty then
37             parts(2).split(",").toList.map(PlayingCard.fromString)
38             else List()
39           val player = new Player(name)
40           player.points = points
41           player.hand = hand
42           players :+= player
43   
44         case line if line.startsWith("TABLE:") =>
45           val cardStrings = line.stripPrefix("TABLE:").split(",").toList.filter(_.nonEmpty)
46           tableCards = cardStrings.map(PlayingCard.fromString)
47   
48         case line if line.startsWith("DECK:") =>
49           val cardStrings = line.stripPrefix("DECK:").split(",").toList.filter(_.nonEmpty)
50           deckCards = cardStrings.map(PlayingCard.fromString)
51   
52         case line if line.startsWith("TURN:") =>
53           turnPlayerName = line.stripPrefix("TURN:")
54   
55         case _ =>
56       }
57       val game = new Game(players)
58       game.board.tableCards = tableCards
59       game.board.deck.setCards(deckCards)
60       val index =players.indexWhere(_.name == turnPlayerName)
61       if index != -1 then game.currentPlayerIndex = index
62       game
